(student_id			INTEGER

,program_name		VARCHAR (50)
,course_id			VARCHAR (8)
,course_name		VARCHAR (50)

,grade				VARCHAR (2)
,instructor_id		INTEGER

,office_location	VARCHAR (50)
);
